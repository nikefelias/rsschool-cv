Codewars Task:

Description: Calculate total points based on match results.


function points(games) {
  let totalPoints = 0;
  for (let i = 0; i < games.length; i++) {
    let match = games[i];
    let scores = match.split(":");
    let x = Number(scores[0]);
    let y = Number(scores[1]);
    if (x > y) {
      totalPoints += 3;
    } else if (x === y) {
      totalPoints += 1;
    }
  }
  return totalPoints;
}
